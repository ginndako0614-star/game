<script>

let players = [];

let deck = [];

let currentPlayer = 0;


function startGame() {

  const count = Number(document.getElementById("playerCount").value);


  players = [];

  for (let i = 1; i <= count; i++) {

    players.push({

      name: "Player " + i,

      point: 500

    });

  }


  deck = [];

  const suits = ["♠", "♥", "♦", "♣"];


  for (let i = 0; i < 13; i++) {

    suits.forEach(suit => {

      deck.push(suit);

    });

  }


  deck.sort(() => Math.random() - 0.5);


  currentPlayer = 0;


  document.getElementById("setup").style.display = "none";

  document.getElementById("game").style.display = "block";

  document.getElementById("result").style.display = "none";


  const bgm = document.getElementById("bgm");

  bgm.volume = 0.5;

  bgm.play().catch(() => {

    console.log("BGMはユーザー操作後に再生されます");

  });


  updateScreen();

}


function guess(selectedSuit) {

  if (deck.length === 0) {

    showResult();

    return;

  }


  const drawnSuit = deck.pop();

  const player = players[currentPlayer];


  if (selectedSuit === drawnSuit) {

    player.point += 100;

    alert(player.name + " 正解！ 出た絵柄は " + drawnSuit + "。+100POINT");

  } else {

    player.point -= 100;

    alert(player.name + " 不正解！ 出た絵柄は " + drawnSuit + "。-100POINT");

  }


  if (deck.length === 0 || player.point <= 0) {

    showResult();

    return;

  }


  currentPlayer++;


  if (currentPlayer >= players.length) {

    currentPlayer = 0;

  }


  updateScreen();

}


function updateScreen() {

  document.getElementById("turnText").textContent =

    players[currentPlayer].name + " の番";


  document.getElementById("deckText").textContent =

    "残り山札：" + deck.length + "枚";


  let html = "";


  players.forEach(player => {

    html += `

      <div class="player">

        ${player.name}：${player.point} POINT

      </div>

    `;

  });


  document.getElementById("players").innerHTML = html;

}


function showResult() {

  document.getElementById("game").style.display = "none";

  document.getElementById("result").style.display = "block";


  const bgm = document.getElementById("bgm");

  bgm.pause();


  const ranking = [...players].sort((a, b) => b.point - a.point);


  let html = "";


  ranking.forEach((player, index) => {

    html += `

      <div class="result-box">

        <h3>${index + 1}位</h3>

        <p>${player.name}</p>

        <p>${player.point} POINT</p>

      </div>

    `;

  });


  document.getElementById("resultText").innerHTML = html;

}
